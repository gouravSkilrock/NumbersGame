package com.skilrock.lms.web.instantWin.reportsMgmt.beans;

import java.io.Serializable;

	private int orgId;
	private double saleAmt;
	private double cancelAmt;
	private double pwtAmt;
	private double pwtDirAmt;

		return orgId;
	}

		this.orgId = orgId;
	}

		return saleAmt;
	}

		this.saleAmt = saleAmt;
	}

	public double getCancelAmt() {
		return cancelAmt;

		return pwtAmt;

		this.pwtAmt = pwtAmt;

		return pwtDirAmt;
